```typescript
// components/FullscreenLayout.tsx (84KB)
export function FullscreenLayout({
  scrollable,
  bottom,
  overlay,
  bottomFloat,
  modal,
  modalScrollRef,
  scrollRef,
  dividerYRef,
  hidePill,
  hideSticky,
  newMessageCount,
  onPillClick,
}) {
  const { columns, rows } = useTerminalSize()
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 计算布局高度
  const bottomHeight = calculateHeight(bottom)
  const scrollableHeight = rows - bottomHeight
  
  // 粘性提示状态
  const [stickyPrompt, setStickyPrompt] = useState<StickyPrompt | null>(null)
  
  // 滚动位置
  const [scrollTop, setScrollTop] = useState(0)
  
  // 检查是否显示 "跳到底部" 按钮
  const showPill = !hidePill && dividerYRef?.current !== null
  
  return (
    <Box flexDirection="column" height={rows}>
      {/* 可滚动区域 */}
      <Box
        flexDirection="column"
        height={scrollableHeight}
        overflow="hidden"
      >
        {/* 粘性提示 */}
        {stickyPrompt && !hideSticky && (
          <Box
            position="sticky"
            top={0}
            backgroundColor={theme.background}
            zIndex={10}
          >
            {stickyPrompt}
          </Box>
        )}
        
        {/* 滚动内容 */}
        <ScrollBox
          ref={scrollRef}
          height={scrollableHeight}
          onScroll={setScrollTop}
        >
          {scrollable}
        </ScrollBox>
        
        {/* 浮动内容 */}
        {bottomFloat && (
          <Box
            position="absolute"
            bottom={bottomHeight + 1}
            right={1}
          >
            {bottomFloat}
          </Box>
        )}
      </Box>
      
      {/* 分隔线 */}
      <Box>
        <Text color={theme.inactive}>
          {'─'.repeat(columns)}
        </Text>
      </Box>
      
      {/* 底部固定区域 */}
      <Box height={bottomHeight}>
        {bottom}
      </Box>
      
      {/* 覆盖层 */}
      {overlay && (
        <Box
          position="absolute"
          bottom={bottomHeight}
          left={0}
          right={0}
          backgroundColor={theme.background}
          borderStyle="single"
          borderColor={theme.inactive}
        >
          {overlay}
        </Box>
      )}
      
      {/* 模态框 */}
      {modal && (
        <Box
          position="absolute"
          bottom={0}
          left={0}
          right={0}
          height={rows}
          backgroundColor={theme.background}
          borderStyle="single"
          borderColor={theme.inactive}
        >
          <ModalContext.Provider value={{ scrollRef: modalScrollRef }}>
            {modal}
          </ModalContext.Provider>
        </Box>
      )}
      
      {/* "跳到底部" 按钮 */}
      {showPill && (
        <Box
          position="absolute"
          bottom={bottomHeight + 2}
          left="50%"
          transform="translateX(-50%)"
        >
          <Box
            backgroundColor={theme.claude}
            borderStyle="round"
            paddingX={1}
            onClick={onPillClick}
          >
            <Text color="white">
              {newMessageCount > 0
                ? `${newMessageCount} new messages`
                : 'Jump to bottom'}
            </Text>
          </Box>
        </Box>
      )}
    </Box>
  )
}
```

## 滚动盒子

### ScrollBox 组件

```typescript
// ink/components/ScrollBox.tsx
export type ScrollBoxHandle = {
  scrollTo: (options: ScrollToOptions) => void
  scrollToTop: () => void
  scrollToBottom: () => void
  getScrollTop: () => number
  getScrollHeight: () => number
  getClientHeight: () => number
}

export type ScrollBoxProps = {
  height: number
  children: React.ReactNode
  onScroll?: (scrollTop: number) => void
  scrollBehavior?: 'smooth' | 'auto'
}

export const ScrollBox = forwardRef<ScrollBoxHandle, ScrollBoxProps>(
  ({ height, children, onScroll, scrollBehavior = 'auto' }, ref) => {
    const [scrollTop, setScrollTop] = useState(0)
    const contentRef = useRef<BoxHandle>(null)
    const scrollHeightRef = useRef(0)
    
    // 暴露滚动方法
    useImperativeHandle(ref, () => ({
      scrollTo: ({ top, behavior = scrollBehavior }) => {
        if (behavior === 'smooth') {
          // 平滑滚动动画
          animateScroll(scrollTop, top, (value) => {
            setScrollTop(value)
            onScroll?.(value)
          })
        } else {
          setScrollTop(top)
          onScroll?.(top)
        }
      },
      scrollToTop: () => {
        setScrollTop(0)
        onScroll?.(0)
      },
      scrollToBottom: () => {
        const maxScroll = Math.max(0, scrollHeightRef.current - height)
        setScrollTop(maxScroll)
        onScroll?.(maxScroll)
      },
      getScrollTop: () => scrollTop,
      getScrollHeight: () => scrollHeightRef.current,
      getClientHeight: () => height,
    }))
    
    // 处理鼠标滚轮
    useEffect(() => {
      const handleWheel = (event: WheelEvent) => {
        const delta = event.deltaY
        const newScrollTop = Math.max(
          0,
          Math.min(scrollTop + delta, scrollHeightRef.current - height)
        )
        setScrollTop(newScrollTop)
        onScroll?.(newScrollTop)
      }
      
      process.stdin.on('wheel', handleWheel)
      return () => {
        process.stdin.off('wheel', handleWheel)
      }
    }, [scrollTop, height, onScroll])
    
    // 处理键盘滚动
    useEffect(() => {
      const handleKey = (event: KeyboardEvent) => {
        const scrollAmount = 3 // 每次滚动 3 行
        
        switch (event.key) {
          case 'PageUp':
            setScrollTop(prev => Math.max(0, prev - height))
            break
          case 'PageDown':
            setScrollTop(prev => 
              Math.min(prev + height, scrollHeightRef.current - height)
            )
            break
          case 'Home':
            setScrollTop(0)
            break
          case 'End':
            setScrollTop(scrollHeightRef.current - height)
            break
        }
      }
      
      process.stdin.on('keypress', handleKey)
      return () => {
        process.stdin.off('keypress', handleKey)
      }
    }, [height])
    
    return (
      <Box
        ref={contentRef}
        height={height}
        overflow="hidden"
        onLayout={(layout) => {
          scrollHeightRef.current = layout.scrollHeight
        }}
      >
        <Box
          flexDirection="column"
          transform={`translateY(-${scrollTop}px)`}
        >
          {children}
        </Box>
      </Box>
    )
  }
)
```

## 虚拟消息列表

### VirtualMessageList 组件

```typescript
// components/VirtualMessageList.tsx (148KB)
export type StickyPrompt = {
  content: React.ReactNode
  height: number
}

export type JumpHandle = {
  jumpToBottom: () => void
  jumpToIndex: (index: number) => void
}

export const VirtualMessageList = forwardRef<JumpHandle, VirtualMessageListProps>(
  ({ messages, renderItem, stickyPrompt, onNewMessage }, ref) => {
    const { columns, rows } = useTerminalSize()
    const scrollRef = useRef<ScrollBoxHandle>(null)
    
    // 计算可见区域
    const containerHeight = rows - 10 // 预留底部空间
    
    // 消息高度缓存
    const heightCache = useRef(new Map<string, number>())
    
    // 计算消息高度
    const estimateItemHeight = useCallback((message: Message) => {
      // 从缓存获取
      if (heightCache.current.has(message.id)) {
        return heightCache.current.get(message.id)!
      }
      
      // 估算高度
      let height = 1 // 基础高度
      
      // 文本消息：计算行数
      if (message.type === 'user' || message.type === 'assistant') {
        const text = message.content[0]?.text || ''
        const lines = text.split('\n').length
        height += Math.ceil(lines / columns)
      }
      
      // 工具使用：额外高度
      if (message.type === 'tool_use') {
        height += 3
      }
      
      // 缓存高度
      heightCache.current.set(message.id, height)
      
      return height
    }, [columns])
    
    // 虚拟滚动
    const {
      visibleItems,
      totalHeight,
      offsetY,
      onScroll,
    } = useVirtualScroll({
      items: messages,
      itemHeight: estimateItemHeight,
      containerHeight,
    })
    
    // 暴露跳转方法
    useImperativeHandle(ref, () => ({
      jumpToBottom: () => {
        scrollRef.current?.scrollToBottom()
      },
      jumpToIndex: (index: number) => {
        // 计算目标位置
        let targetY = 0
        for (let i = 0; i < index; i++) {
          targetY += estimateItemHeight(messages[i])
        }
        scrollRef.current?.scrollTo({ top: targetY })
      },
    }))
    
    // 监听新消息
    useEffect(() => {
      if (onNewMessage) {
        // 自动滚动到底部
        scrollRef.current?.scrollToBottom()
      }
    }, [messages.length, onNewMessage])
    
    return (
      <ScrollBox
        ref={scrollRef}
        height={containerHeight}
        onScroll={onScroll}
      >
        <Box flexDirection="column">
          {/* 粘性提示 */}
          {stickyPrompt && (
            <Box
              position="sticky"
              top={0}
              height={stickyPrompt.height}
              zIndex={10}
            >
              {stickyPrompt.content}
            </Box>
          )}
          
          {/* 偏移填充 */}
          <Box height={offsetY} />
          
          {/* 可见消息 */}
          {visibleItems.map((message, index) => (
            <Box key={message.id}>
              {renderItem(message, index)}
            </Box>
          ))}
        </Box>
      </ScrollBox>
    )
  }
)
```

## 虚拟滚动 Hook

### useVirtualScroll

```typescript
// hooks/useVirtualScroll.ts
export type UseVirtualScrollOptions<T> = {
  items: T[]
  itemHeight: (item: T) => number
  containerHeight: number
  overscan?: number
}

export type UseVirtualScrollResult<T> = {
  visibleItems: T[]
  totalHeight: number
  offsetY: number
  onScroll: (scrollTop: number) => void
}

export function useVirtualScroll<T>({
  items,
  itemHeight,
  containerHeight,
  overscan = 5,
}: UseVirtualScrollOptions<T>): UseVirtualScrollResult<T> {
  const [scrollTop, setScrollTop] = useState(0)
  
  // 计算总高度
  const totalHeight = useMemo(() => {
    return items.reduce((sum, item) => sum + itemHeight(item), 0)
  }, [items, itemHeight])
  
  // 计算可见范围
  const { startIndex, endIndex, offsetY } = useMemo(() => {
    let currentY = 0
    let startIndex = 0
    let endIndex = items.length - 1
    let offsetY = 0
    
    // 找到起始索引
    for (let i = 0; i < items.length; i++) {
      const height = itemHeight(items[i])
      if (currentY + height > scrollTop) {
        startIndex = Math.max(0, i - overscan)
        offsetY = currentY
        break
      }
      currentY += height
    }
    
    // 找到结束索引
    currentY = offsetY
    for (let i = startIndex; i < items.length; i++) {
      const height = itemHeight(items[i])
      if (currentY > scrollTop + containerHeight) {
        endIndex = Math.min(items.length - 1, i + overscan)
        break
      }
      currentY += height
    }
    
    return { startIndex, endIndex, offsetY }
  }, [items, itemHeight, scrollTop, containerHeight, overscan])
  
  // 获取可见项
  const visibleItems = useMemo(() => {
    return items.slice(startIndex, endIndex + 1)
  }, [items, startIndex, endIndex])
  
  // 处理滚动
  const onScroll = useCallback((newScrollTop: number) => {
    setScrollTop(newScrollTop)
  }, [])
  
  return {
    visibleItems,
    totalHeight,
    offsetY,
    onScroll,
  }
}
```

```typescript
// components/design-system/Dialog.tsx
export function Dialog({
  title,
  children,
  onClose,
  width = 60,
  height,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  const { columns, rows } = useTerminalSize()
  
  // 计算居中位置
  const left = Math.floor((columns - width) / 2)
  const top = Math.floor((rows - (height || 20)) / 2)
  
  return (
    <Box
      position="absolute"
      left={left}
      top={top}
      width={width}
      height={height}
      flexDirection="column"
      borderStyle="double"
      borderColor={theme.claude}
      backgroundColor={theme.background}
    >
      {/* 标题栏 */}
      <Box
        backgroundColor={theme.claude}
        paddingX={1}
        justifyContent="space-between"
      >
        <Text color="white" bold>
          {title}
        </Text>
        <Text
          color="white"
          onClick={onClose}
          cursor="pointer"
        >
          ✕
        </Text>
      </Box>
      
      {/* 内容 */}
      <Box
        flexDirection="column"
        padding={1}
        overflow="hidden"
      >
        {children}
      </Box>
    </Box>
  )
}
```

```typescript
// components/ui/TreeSelect.tsx (38KB)
export type TreeNode = {
  id: string
  label: string
  children?: TreeNode[]
  icon?: string
  data?: any
}

export function TreeSelect({
  nodes,
  selectedId,
  onSelect,
  onExpand,
  expandedIds,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 渲染树节点
  const renderNode = (node: TreeNode, depth: number = 0) => {
    const isSelected = node.id === selectedId
    const isExpanded = expandedIds.has(node.id)
    const hasChildren = node.children && node.children.length > 0
    
    return (
      <Box key={node.id} flexDirection="column">
        {/* 节点行 */}
        <Box
          paddingLeft={depth * 2}
          backgroundColor={
            isSelected ? theme.selectionBg : undefined
          }
          onClick={() => onSelect(node.id)}
          cursor="pointer"
        >
          {/* 展开/折叠图标 */}
          {hasChildren ? (
            <Text
              onClick={(e) => {
                e.stopPropagation()
                onExpand(node.id)
              }}
              cursor="pointer"
            >
              {isExpanded ? '▼' : '▶'}
            </Text>
          ) : (
            <Text> </Text>
          )}
          
          {/* 图标 */}
          {node.icon && (
            <Text> {node.icon}</Text>
          )}
          
          {/* 标签 */}
          <Text
            color={isSelected ? theme.claude : theme.text}
            bold={isSelected}
          >
            {node.label}
          </Text>
        </Box>
        
        {/* 子节点 */}
        {isExpanded && hasChildren && (
          <Box flexDirection="column">
            {node.children.map(child =>
              renderNode(child, depth + 1)
            )}
          </Box>
        )}
      </Box>
    )
  }
  
  return (
    <Box flexDirection="column">
      {nodes.map(node => renderNode(node))}
    </Box>
  )
}
```

```typescript
// components/design-system/FuzzyPicker.tsx (40KB)
export function FuzzyPicker({
  items,
  onSelect,
  onClose,
  placeholder = 'Search...',
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  const [query, setQuery] = useState('')
  const [selectedIndex, setSelectedIndex] = useState(0)
  
  // 模糊匹配
  const filteredItems = useMemo(() => {
    if (!query) return items
    
    return items.filter(item =>
      fuzzyMatch(item.label, query)
    ).sort((a, b) =>
      fuzzyScore(b.label, query) - fuzzyScore(a.label, query)
    )
  }, [items, query])
  
  // 处理键盘事件
  useInput((input, key) => {
    if (key.escape) {
      onClose()
    } else if (key.return) {
      if (filteredItems[selectedIndex]) {
        onSelect(filteredItems[selectedIndex])
      }
    } else if (key.upArrow) {
      setSelectedIndex(prev => Math.max(0, prev - 1))
    } else if (key.downArrow) {
      setSelectedIndex(prev =>
        Math.min(filteredItems.length - 1, prev + 1)
      )
    }
  })
  
  return (
    <Dialog
      title="Search"
      onClose={onClose}
      width={60}
      height={20}
    >
      {/* 搜索输入 */}
      <Box
        borderStyle="single"
        borderColor={theme.inactive}
        paddingX={1}
        marginBottom={1}
      >
        <Text color={theme.subtle}>{placeholder}</Text>
        <Text>{query}</Text>
        <Text inverse> </Text> {/* 光标 */}
      </Box>
      
      {/* 结果列表 */}
      <Box flexDirection="column" overflow="hidden">
        {filteredItems.map((item, index) => (
          <Box
            key={item.id}
            backgroundColor={
              index === selectedIndex
                ? theme.selectionBg
                : undefined
            }
            paddingX={1}
            onClick={() => onSelect(item)}
            cursor="pointer"
          >
            <Text
              color={
                index === selectedIndex
                  ? theme.claude
                  : theme.text
              }
              bold={index === selectedIndex}
            >
              {item.icon && `${item.icon} `}
              {item.label}
            </Text>
            {item.description && (
              <Text color={theme.subtle} dimColor>
                {' '}{item.description}
              </Text>
            )}
          </Box>
        ))}
      </Box>
      
      {/* 提示 */}
      <Box marginTop={1}>
        <Text color={theme.subtle} dimColor>
          ↑↓ Navigate • Enter Select • Esc Cancel
        </Text>
      </Box>
    </Dialog>
  )
}
```

```typescript
export function calculateHeight(node: React.ReactNode): number {
  // 从 Box 的 height 属性获取
  if (React.isValidElement(node) && node.props.height) {
    return node.props.height as number
  }
  
  // 默认高度
```
