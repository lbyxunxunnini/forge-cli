# 状态机增强

> 支持严格的状态转换验证、完整的状态历史记录、崩溃后精确恢复

## 1. 概述

状态机增强系统提供了完整的状态管理能力，包括状态转换验证、历史记录、快照恢复等功能。

### 核心功能

- **状态转换验证**：严格的状态转换规则，防止非法状态转换
- **历史记录**：完整的状态变更历史，支持回溯和审计
- **快照恢复**：支持保存和恢复状态机快照，实现崩溃恢复
- **转换图可视化**：可视化状态转换关系

## 2. 基本使用

### 创建状态机

```typescript
import { StateMachine, createStateMachine } from './cli/state-machine.js'

// 创建状态机
const stateMachine = createStateMachine({
  initialState: 'idle',
  states: ['idle', 'processing', 'completed', 'error'],
  transitions: [
    { from: 'idle', to: 'processing', event: 'start' },
    { from: 'processing', to: 'completed', event: 'finish' },
    { from: 'processing', to: 'error', event: 'fail' },
    { from: 'error', to: 'idle', event: 'reset' },
    { from: 'completed', to: 'idle', event: 'reset' },
  ],
})
```

### 触发事件

```typescript
// 触发事件
stateMachine.send('start')  // idle -> processing
stateMachine.send('finish') // processing -> completed
stateMachine.send('reset')  // completed -> idle
```

### 获取状态

```typescript
// 获取当前状态
const currentState = stateMachine.getCurrentState()
console.log(currentState)  // 'idle'

// 获取状态历史
const history = stateMachine.getHistory()
console.log(history)

// 获取上下文
const context = stateMachine.getContext()
console.log(context)
```

## 3. 配置选项

### 状态机配置

```typescript
interface StateMachineConfig {
  initialState: StateType          // 初始状态
  states: StateType[]              // 所有状态列表
  transitions: StateTransition[]   // 状态转换规则
  onStateChange?: (from: StateType, to: StateType, event: string) => void
  onError?: (error: StateMachineError) => void
}
```

### 状态转换规则

```typescript
interface StateTransition {
  from: StateType                  // 源状态
  to: StateType                    // 目标状态
  event: string                    // 触发事件
  guard?: (context: any) => boolean  // 转换条件
  action?: (context: any) => void    // 转换动作
}
```

### 示例：带条件的转换

```typescript
const stateMachine = createStateMachine({
  initialState: 'idle',
  states: ['idle', 'processing', 'completed', 'error'],
  transitions: [
    {
      from: 'idle',
      to: 'processing',
      event: 'start',
      guard: (context) => context.canStart === true,  // 条件检查
      action: (context) => console.log('开始处理'),    // 转换动作
    },
    {
      from: 'processing',
      to: 'completed',
      event: 'finish',
      guard: (context) => context.isComplete === true,
    },
  ],
})

// 设置上下文
stateMachine.setContext({ canStart: true })

// 触发事件（会检查条件）
stateMachine.send('start')
```

## 5. 快照恢复

### 创建快照

```typescript
// 创建快照
const snapshot = stateMachine.createSnapshot()

// 快照结构
interface StateMachineSnapshot {
  currentState: StateType
  history: StateHistoryEntry[]
  context: any
  timestamp: number
}
```

### 保存快照

```typescript
import { saveSnapshot } from './cli/state-machine.js'

// 保存到文件
await saveSnapshot(snapshot, 'state-snapshot.json')
```

### 恢复快照

```typescript
import { loadSnapshot } from './cli/state-machine.js'

// 从文件加载
const snapshot = await loadSnapshot('state-snapshot.json')
if (snapshot) {
  stateMachine.restoreFromSnapshot(snapshot)
}
```

### 崩溃恢复示例

```typescript
import { createStateMachine, saveSnapshot, loadSnapshot } from './cli/state-machine.js'

// 创建状态机
const stateMachine = createStateMachine({
  initialState: 'idle',
  states: ['idle', 'processing', 'completed', 'error'],
  transitions: [
    { from: 'idle', to: 'processing', event: 'start' },
    { from: 'processing', to: 'completed', event: 'finish' },
  ],
})

// 定期保存快照
setInterval(() => {
  const snapshot = stateMachine.createSnapshot()
  saveSnapshot(snapshot, 'auto-snapshot.json')
}, 60000)  // 每分钟保存一次

// 程序重启后恢复
async function recover() {
  const snapshot = await loadSnapshot('auto-snapshot.json')
  if (snapshot) {
    stateMachine.restoreFromSnapshot(snapshot)
    console.log('状态已恢复:', stateMachine.getCurrentState())
  }
}
```

## 8. 完整示例

```typescript
import { createStateMachine, saveSnapshot, loadSnapshot } from './cli/state-machine.js'

// 定义状态机
const config = {
  initialState: 'idle',
  states: ['idle', 'connecting', 'connected', 'error', 'disconnected'],
  transitions: [
    { from: 'idle', to: 'connecting', event: 'connect' },
    { from: 'connecting', to: 'connected', event: 'success' },
    { from: 'connecting', to: 'error', event: 'fail' },
    { from: 'connected', to: 'disconnected', event: 'disconnect' },
    { from: 'error', to: 'idle', event: 'reset' },
    { from: 'disconnected', to: 'idle', event: 'reset' },
  ],
  onStateChange: (from, to, event) => {
    console.log(`状态变更: ${from} → ${to} (${event})`)
  },
  onError: (error) => {
    console.error(`状态机错误: ${error.message}`)
  },
}

// 创建状态机
const stateMachine = createStateMachine(config)

// 使用状态机
async function main() {
  // 显示初始状态
  console.log('初始状态:', stateMachine.getCurrentState())
  console.log(stateMachine.render())

  // 触发事件
  stateMachine.send('connect')
  stateMachine.send('success')

  // 设置上下文
  stateMachine.setContext({ userId: '123', sessionId: 'abc' })

  // 保存快照
  const snapshot = stateMachine.createSnapshot()
  await saveSnapshot(snapshot, 'connection-state.json')

  // 模拟崩溃恢复
  const recoveredSnapshot = await loadSnapshot('connection-state.json')
  if (recoveredSnapshot) {
    stateMachine.restoreFromSnapshot(recoveredSnapshot)
    console.log('恢复后的状态:', stateMachine.getCurrentState())
  }

  // 显示历史
  console.log(stateMachine.render())
}

main().catch(console.error)
```

## 9. 最佳实践

### 1. 状态定义

```typescript
// 使用常量定义状态
const STATES = {
  IDLE: 'idle',
  PROCESSING: 'processing',
  COMPLETED: 'completed',
  ERROR: 'error',
} as const

const stateMachine = createStateMachine({
  initialState: STATES.IDLE,
  states: Object.values(STATES),
  transitions: [
    { from: STATES.IDLE, to: STATES.PROCESSING, event: 'start' },
    // ...
  ],
})
```

### 2. 条件检查

```typescript
// 使用 guard 进行条件检查
const transitions = [
  {
    from: 'idle',
    to: 'processing',
    event: 'start',
    guard: (context) => {
      // 检查前置条件
      return context.hasPermission && context.isConnected
    },
  },
]
```

### 3. 错误处理

```typescript
// 处理状态机错误
const stateMachine = createStateMachine({
  // ...
  onError: (error) => {
    switch (error.code) {
      case 'INVALID_TRANSITION':
        console.log('无效的状态转换')
        break
      case 'GUARD_FAILED':
        console.log('转换条件不满足')
        break
    }
  },
})

// 安全地触发事件
try {
  stateMachine.send('start')
} catch (error) {
  if (error instanceof StateMachineError) {
    console.log(`错误: ${error.message}`)
  }
}
```

### 4. 定期快照

```typescript
// 定期保存快照
const snapshotInterval = setInterval(() => {
  const snapshot = stateMachine.createSnapshot()
  saveSnapshot(snapshot, 'auto-snapshot.json')
    .catch(console.error)
}, 60000)

// 程序退出时保存
process.on('SIGINT', async () => {
  clearInterval(snapshotInterval)
  const snapshot = stateMachine.createSnapshot()
  await saveSnapshot(snapshot, 'final-snapshot.json')
  process.exit(0)
})
```
