```typescript
// components/Spinner/SpinnerGlyph.tsx
export function SpinnerGlyph({
  frame,
  messageColor,
  stalledIntensity = 0,
  reducedMotion = false,
  time = 0,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 减少动画模式：缓慢闪烁的圆点
  if (reducedMotion) {
    const isDim = Math.floor(time / (REDUCED_MOTION_CYCLE_MS / 2)) % 2 === 1
    return (
      <Box flexWrap="wrap" height={1} width={2}>
        <Text color={messageColor} dimColor={isDim}>
          {REDUCED_MOTION_DOT}
        </Text>
      </Box>
    )
  }
  
  // 获取动画帧
  const spinnerChar = SPINNER_FRAMES[frame % SPINNER_FRAMES.length]
  
  // 卡住状态：颜色渐变到红色
  if (stalledIntensity > 0) {
    const baseColorStr = theme[messageColor]
    const baseRGB = baseColorStr ? parseRGB(baseColorStr) : null
    
    if (baseRGB) {
      const interpolated = interpolateColor(baseRGB, ERROR_RED, stalledIntensity)
      return (
        <Box flexWrap="wrap" height={1} width={2}>
          <Text color={toRGBColor(interpolated)}>
            {spinnerChar}
          </Text>
        </Box>
      )
    }
    
    // ANSI 主题回退
    const color = stalledIntensity > 0.5 ? 'error' : messageColor
    return (
      <Box flexWrap="wrap" height={1} width={2}>
        <Text color={color}>
          {spinnerChar}
        </Text>
      </Box>
    )
  }
  
  // 正常状态
  return (
    <Box flexWrap="wrap" height={1} width={2}>
      <Text color={messageColor}>
        {spinnerChar}
      </Text>
    </Box>
  )
}
```

```typescript
// constants/spinnerVerbs.ts
export const SPINNER_VERBS = [
  'Accomplishing',
  'Actioning',
  'Actualizing',
  'Architecting',
  'Baking',
  'Beaming',
  // ... 200+ 动词
  'Envisioning',
  'Generating',
  'Thinking',
  'Working',
  // ...
]

export function getSpinnerVerbs(): string[] {
  const settings = getInitialSettings()
  const config = settings.spinnerVerbs
  
  if (!config) return SPINNER_VERBS
  
  if (config.mode === 'replace') {
    return config.verbs.length > 0 ? config.verbs : SPINNER_VERBS
  }
  
  return [...SPINNER_VERBS, ...config.verbs]
}
```

## 进度显示

### Agent 进度行

```typescript
// components/AgentProgressLine.tsx
export function AgentProgressLine({
  agentType,
  description,
  name,
  descriptionColor,
  taskDescription,
  toolUseCount,
  tokens,
  color,
  isLast,
  isResolved,
  isError,
  isAsync = false,
  shouldAnimate,
  lastToolInfo,
  hideType = false,
}) {
  // 树形结构字符
  const treeChar = isLast ? '└─' : '├─'
  
  // 后台任务状态
  const isBackgrounded = isAsync && isResolved
  
  // 获取状态文本
  const getStatusText = () => {
    if (!isResolved) {
      return lastToolInfo || 'Initializing…'
    }
    if (isBackgrounded) {
      return taskDescription ?? 'Running in the background'
    }
    return 'Done'
  }
  
  return (
    <Box>
      {/* 树形连接符 */}
      <Text dimColor>{treeChar} </Text>
      
      {/* Agent 类型 */}
      {hideType ? (
        <Text bold>{name ?? description ?? agentType}</Text>
      ) : (
        <Text bold backgroundColor={color} color={color ? 'inverseText' : undefined}>
          {agentType}
        </Text>
      )}
      
      {/* 描述 */}
      {description && (
        <Text dimColor> ({description})</Text>
      )}
      
      {/* 状态文本 */}
      <Text color={isResolved ? 'success' : 'subtle'}>
        {' '}{getStatusText()}
      </Text>
      
      {/* 工具使用计数 */}
      {toolUseCount > 0 && (
        <Text dimColor> [{toolUseCount} tools]</Text>
      )}
      
      {/* Token 计数 */}
      {tokens !== null && (
        <Text dimColor> [{formatNumber(tokens)} tokens]</Text>
      )}
    </Box>
  )
}
```

### Shell 进度消息

```typescript
// components/shell/ShellProgressMessage.tsx
export function ShellProgressMessage({
  output,
  fullOutput,
  elapsedTimeSeconds,
  totalLines,
  totalBytes,
  timeoutMs,
  verbose,
}) {
  // 处理输出
  const strippedFullOutput = stripAnsi(fullOutput.trim())
  const strippedOutput = stripAnsi(output.trim())
  const lines = strippedOutput.split('\n').filter(Boolean)
  
  // 显示最后 5 行（非 verbose 模式）
  const displayLines = verbose ? strippedFullOutput : lines.slice(-5).join('\n')
  
  // 计算额外行数
  const extraLines = totalLines ? Math.max(0, totalLines - 5) : 0
  
  // 构建行状态
  let lineStatus = ''
  if (!verbose && totalBytes && totalLines) {
    lineStatus = `~${totalLines} lines`
  } else if (!verbose && extraLines > 0) {
    lineStatus = `+${extraLines} lines`
  }
  
  return (
    <MessageResponse>
      <OffscreenFreeze>
        {/* 运行状态 */}
        <Text dimColor>Running… </Text>
        
        {/* 计时器 */}
        <ShellTimeDisplay
          elapsedTimeSeconds={elapsedTimeSeconds}
          timeoutMs={timeoutMs}
        />
        
        {/* 输出内容 */}
        {displayLines && (
          <Box flexDirection="column">
            <Text>{displayLines}</Text>
          </Box>
        )}
        
        {/* 行状态 */}
        {lineStatus && (
          <Text dimColor> ({lineStatus})</Text>
        )}
      </OffscreenFreeze>
    </MessageResponse>
  )
}
```

### Shell 计时器显示

```typescript
// components/shell/ShellTimeDisplay.tsx
export function ShellTimeDisplay({
  elapsedTimeSeconds,
  timeoutMs,
}) {
  // 格式化时间
  const formatTime = (seconds: number) => {
    if (seconds < 60) {
      return `${Math.round(seconds)}s`
    }
    const minutes = Math.floor(seconds / 60)
    const remainingSeconds = Math.round(seconds % 60)
    return `${minutes}m ${remainingSeconds}s`
  }
  
  // 计算超时进度
  const timeoutProgress = timeoutMs
    ? (elapsedTimeSeconds * 1000) / timeoutMs
    : null
  
  return (
    <Text dimColor>
      ({formatTime(elapsedTimeSeconds)})
      {timeoutProgress !== null && (
        <Text color={timeoutProgress > 0.8 ? 'error' : 'subtle'}>
          {' '}{Math.round(timeoutProgress * 100)}%
        </Text>
      )}
    </Text>
  )
}
```

```typescript
// ink/hooks/useAnimationFrame.ts
export function useAnimationFrame(intervalMs: number): number {
  const [frame, setFrame] = useState(0)
  
  useEffect(() => {
    const interval = setInterval(() => {
      setFrame(prev => prev + 1)
    }, intervalMs)
    
    return () => clearInterval(interval)
  }, [intervalMs])
  
  return frame
}
```

```typescript
// hooks/useElapsedTime.ts
export function useElapsedTime(startTimeRef: RefObject<number>) {
  const [elapsed, setElapsed] = useState(0)
  
  useAnimationFrame(1000)
  
  useEffect(() => {
    if (startTimeRef.current) {
      setElapsed((Date.now() - startTimeRef.current) / 1000)
    }
  })
  
  return elapsed
}
```

```typescript
// components/Spinner/useShimmerAnimation.ts
export function useShimmerAnimation(
  message: string,
  intervalMs: number = SHIMMER_INTERVAL_MS
) {
  const [segments, setSegments] = useState<ShimmerSegment[]>([])
  
  useEffect(() => {
    const interval = setInterval(() => {
      setSegments(computeShimmerSegments(message))
    }, intervalMs)
    
    return () => clearInterval(interval)
  }, [message, intervalMs])
  
  return segments
}
```
