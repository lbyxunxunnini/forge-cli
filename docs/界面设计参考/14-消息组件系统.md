```typescript
// components/MessageRow.tsx
export function MessageRow({
  message,
  verbose,
  width,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 根据消息类型渲染
  switch (message.type) {
    case 'user':
      return <UserTextMessage message={message} />
    case 'assistant':
      return <AssistantTextMessage message={message} />
    case 'system':
      return <SystemTextMessage message={message} />
    case 'tool_use':
      return <AssistantToolUseMessage message={message} />
    case 'tool_result':
      return <UserToolResultMessage message={message} />
    default:
      return null
  }
}
```

## AI 消息组件

### AI 文本消息

```typescript
// components/messages/AssistantTextMessage.tsx
export function AssistantTextMessage({
  param,
  addMargin,
  shouldShowDot,
  verbose,
  onOpenRateLimitOptions,
}) {
  const { text } = param
  const isSelected = useContext(MessageActionsSelectedContext)
  
  // 空消息处理
  if (isEmptyMessageText(text)) {
    return null
  }
  
  // 速率限制错误
  if (isRateLimitErrorMessage(text)) {
    return (
      <RateLimitMessage
        text={text}
        onOpenRateLimitOptions={onOpenRateLimitOptions}
      />
    )
  }
  
  // 特殊消息处理
  switch (text) {
    case NO_RESPONSE_REQUESTED:
      return null
    case PROMPT_TOO_LONG_ERROR_MESSAGE:
      return <TokenWarning />
    case API_TIMEOUT_ERROR_MESSAGE:
      return <ErrorMessage text={text} />
    // ... 更多特殊消息
  }
  
  return (
    <MessageResponse>
      <Box marginY={addMargin ? 1 : 0}>
        {/* 消息点指示器 */}
        {shouldShowDot && (
          <Text color="claude">⏺ </Text>
        )}
        
        {/* Markdown 内容 */}
        <Markdown
          content={text}
          verbose={verbose}
        />
        
        {/* 中断指示器 */}
        {text.includes(INTERRUPTED_INDICATOR) && (
          <InterruptedByUser />
        )}
      </Box>
    </MessageResponse>
  )
}
```

```typescript
// components/messages/SystemTextMessage.tsx
export function SystemTextMessage({
  message,
  verbose,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  const text = message.content[0].text
  
  // 系统消息类型判断
  const isNotice = text.startsWith('⏺')
  const isDivider = text.startsWith('─')
  const isStatus = text.startsWith('*')
  
  return (
    <Box marginY={isDivider ? 0 : 1}>
      {isNotice ? (
        <Text color={theme.subtle}>{text}</Text>
      ) : isDivider ? (
        <Text color={theme.inactive}>{text}</Text>
      ) : isStatus ? (
        <Text color={theme.subtle} dimColor>{text}</Text>
      ) : (
        <Text>{text}</Text>
      )}
    </Box>
  )
}
```

## Markdown 渲染

### Markdown 组件

```typescript
// components/Markdown.tsx
export function Markdown({
  content,
  verbose,
  width,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 解析 Markdown
  const parsed = parseMarkdown(content)
  
  return (
    <Box flexDirection="column">
      {parsed.map((block, index) => {
        switch (block.type) {
          case 'paragraph':
            return (
              <Text key={index}>
                {renderInline(block.content)}
              </Text>
            )
          case 'heading':
            return (
              <Text key={index} bold>
                {block.content}
              </Text>
            )
          case 'code':
            return (
              <HighlightedCode
                key={index}
                code={block.content}
                language={block.language}
              />
            )
          case 'list':
            return (
              <Box key={index} flexDirection="column">
                {block.items.map((item, i) => (
                  <Box key={i}>
                    <Text>• </Text>
                    <Text>{item}</Text>
                  </Box>
                ))}
              </Box>
            )
          case 'blockquote':
            return (
              <Box key={index} marginLeft={2}>
                <Text color={theme.subtle}>
                  ▎ {block.content}
                </Text>
              </Box>
            )
          default:
            return (
              <Text key={index}>{block.content}</Text>
            )
        }
      })}
    </Box>
  )
}
```

### 代码高亮

```typescript
// components/HighlightedCode.tsx
export function HighlightedCode({
  code,
  language,
  showLineNumbers = true,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 语法高亮
  const highlighted = highlightSyntax(code, language)
  
  // 分割行
  const lines = highlighted.split('\n')
  
  return (
    <Box
      backgroundColor={theme.background}
      borderStyle="single"
      borderColor={theme.inactive}
      paddingX={1}
      marginY={1}
    >
      <Box flexDirection="column">
        {lines.map((line, index) => (
          <Box key={index}>
            {/* 行号 */}
            {showLineNumbers && (
              <Text color={theme.inactive} dimColor>
                {String(index + 1).padStart(3)} │
              </Text>
            )}
            {/* 代码内容 */}
            <Text>{line}</Text>
          </Box>
        ))}
      </Box>
    </Box>
  )
}
```

### 虚拟滚动实现

```typescript
// components/VirtualMessageList.tsx
export function VirtualMessageList({
  messages,
  renderItem,
  scrollRef,
  stickyPrompt,
}) {
  const { columns, rows } = useTerminalSize()
  
  // 计算可见区域
  const containerHeight = rows - 10 // 预留底部空间
  
  // 使用虚拟滚动
  const {
    visibleItems,
    totalHeight,
    offsetY,
    onScroll,
  } = useVirtualScroll({
    items: messages,
    itemHeight: estimateItemHeight,
    containerHeight,
  })
  
  return (
    <ScrollBox
      ref={scrollRef}
      height={containerHeight}
      onScroll={onScroll}
    >
      <Box flexDirection="column" paddingTop={offsetY}>
        {/* 粘性提示 */}
        {stickyPrompt && (
          <Box position="sticky" top={0}>
            {stickyPrompt}
          </Box>
        )}
        
        {/* 可见消息 */}
        {visibleItems.map((message, index) => (
          <Box key={message.id}>
            {renderItem(message, index)}
          </Box>
        ))}
      </Box>
    </ScrollBox>
  )
}
```
