```typescript
// components/PromptInput/PromptInput.tsx (355KB)
export function PromptInput({
  onSubmit,
  onCancel,
  onHistoryPrevious,
  onHistoryNext,
  mode,
  vimMode,
  isLoading,
  verbose,
  // ... 更多属性
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 输入状态
  const [input, setInput] = useState('')
  const [cursorPosition, setCursorPosition] = useState(0)
  const [historyIndex, setHistoryIndex] = useState(-1)
  
  // 自动补全状态
  const [suggestions, setSuggestions] = useState([])
  const [selectedSuggestion, setSelectedSuggestion] = useState(0)
  const [showSuggestions, setShowSuggestions] = useState(false)
  
  // 处理输入
  const handleInput = useCallback((text: string) => {
    setInput(prev => {
      const newInput = 
        prev.slice(0, cursorPosition) + 
        text + 
        prev.slice(cursorPosition)
      setCursorPosition(cursorPosition + text.length)
      return newInput
    })
    
    // 更新自动补全
    updateSuggestions(input + text)
  }, [cursorPosition, input])
  
  // 处理特殊键
  const handleKeyPress = useCallback((key: Key) => {
    switch (key.name) {
      case 'return':
        if (showSuggestions && suggestions[selectedSuggestion]) {
          applySuggestion(suggestions[selectedSuggestion])
        } else if (input.trim()) {
          onSubmit(input)
          setInput('')
          setCursorPosition(0)
        }
        break
        
      case 'backspace':
        if (cursorPosition > 0) {
          setInput(prev => 
            prev.slice(0, cursorPosition - 1) + 
            prev.slice(cursorPosition)
          )
          setCursorPosition(cursorPosition - 1)
        }
        break
        
      case 'delete':
        if (cursorPosition < input.length) {
          setInput(prev => 
            prev.slice(0, cursorPosition) + 
            prev.slice(cursorPosition + 1)
          )
        }
        break
        
      case 'left':
        setCursorPosition(Math.max(0, cursorPosition - 1))
        break
        
      case 'right':
        setCursorPosition(Math.min(input.length, cursorPosition + 1))
        break
        
      case 'up':
        if (showSuggestions) {
          setSelectedSuggestion(prev => 
            Math.max(0, prev - 1)
          )
        } else {
          onHistoryPrevious()
        }
        break
        
      case 'down':
        if (showSuggestions) {
          setSelectedSuggestion(prev => 
            Math.min(suggestions.length - 1, prev + 1)
          )
        } else {
          onHistoryNext()
        }
        break
        
      case 'tab':
        if (suggestions[selectedSuggestion]) {
          applySuggestion(suggestions[selectedSuggestion])
        }
        break
        
      case 'escape':
        if (showSuggestions) {
          setShowSuggestions(false)
        } else {
          onCancel()
        }
        break
    }
  }, [
    input, 
    cursorPosition, 
    showSuggestions, 
    suggestions, 
    selectedSuggestion,
    onSubmit,
    onCancel,
    onHistoryPrevious,
    onHistoryNext,
  ])
  
  return (
    <Box flexDirection="column">
      {/* 输入区域 */}
      <Box
        borderStyle="round"
        borderColor={isLoading ? theme.inactive : theme.promptBorder}
        paddingX={1}
      >
        {/* 提示符 */}
        <Text color={theme.claude}>❯ </Text>
        
        {/* 输入文本 */}
        <Box>
          <Text>{input.slice(0, cursorPosition)}</Text>
          <Text inverse> </Text> {/* 光标 */}
          <Text>{input.slice(cursorPosition)}</Text>
        </Box>
      </Box>
      
      {/* 自动补全建议 */}
      {showSuggestions && suggestions.length > 0 && (
        <PromptInputFooterSuggestions
          suggestions={suggestions}
          selectedSuggestion={selectedSuggestion}
          onSelect={applySuggestion}
        />
      )}
      
      {/* 底部栏 */}
      <PromptInputFooter
        mode={mode}
        vimMode={vimMode}
        isLoading={isLoading}
        suggestions={suggestions}
        selectedSuggestion={selectedSuggestion}
      />
    </Box>
  )
}
```

```typescript
export function PromptInputFooterSuggestions({
  suggestions,
  selectedSuggestion,
  onSelect,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  // 获取类型图标
  const getTypeIcon = (type: string) => {
    switch (type) {
      case 'command': return '⌘'
      case 'file': return '📄'
      case 'history': return '🕐'
      case 'skill': return '⚡'
      default: return '•'
    }
  }
  
  return (
    <Box
      backgroundColor={theme.background}
      borderStyle="single"
      borderColor={theme.inactive}
      flexDirection="column"
      paddingX={1}
    >
      {suggestions.map((suggestion, index) => (
        <Box
          key={index}
          backgroundColor={
            index === selectedSuggestion
              ? theme.selectionBg
              : undefined
          }
        >
          {/* 类型图标 */}
          <Text color={theme.subtle}>
            {getTypeIcon(suggestion.type)}
          </Text>
          
          {/* 标签 */}
          <Text
            color={
              index === selectedSuggestion
                ? theme.claude
                : theme.text
            }
            bold={index === selectedSuggestion}
          >
            {suggestion.label}
          </Text>
          
          {/* 描述 */}
          {suggestion.description && (
            <Text color={theme.subtle} dimColor>
              {' '}{suggestion.description}
            </Text>
          )}
        </Box>
      ))}
    </Box>
  )
}
```

```typescript
export function KeybindingProvider({ children }) {

  useEffect(() => {
    setBindings(DEFAULT_BINDINGS)

    loadUserBindings().then(userBindings => {

  return (
    <KeybindingContext.Provider value={{ bindings }}>
      {children}
    </KeybindingContext.Provider>
  )
}
```

### 快捷键 Hook

```typescript
// keybindings/useKeybinding.ts
export function useKeybinding(
  shortcut: string,
  action: () => void,
  context?: string,
) {
  const { bindings } = useContext(KeybindingContext)
  
  useEffect(() => {
    const handler = (event: KeyboardEvent) => {
      // 检查快捷键匹配
      if (matchesShortcut(event, shortcut)) {
        // 检查上下文
        if (!context || isInContext(context)) {
          event.preventDefault()
          action()
        }
      }
    }
    
    // 注册事件监听
    process.stdin.on('keypress', handler)
    
    return () => {
      process.stdin.off('keypress', handler)
    }
  }, [shortcut, action, context, bindings])
}
```

## 历史记录

### 历史记录管理

```typescript
// history.ts
export type HistoryEntry = {
  input: string
  timestamp: number
  sessionId: string
}

export class HistoryManager {
  private entries: HistoryEntry[] = []
  private maxLength: number = 1000
  
  add(input: string, sessionId: string) {
    this.entries.push({
      input,
      timestamp: Date.now(),
      sessionId,
    })
    
    // 限制长度
    if (this.entries.length > this.maxLength) {
      this.entries.shift()
    }
    
    // 保存到文件
    this.save()
  }
  
  search(query: string): HistoryEntry[] {
    return this.entries.filter(entry =>
      entry.input.toLowerCase().includes(query.toLowerCase())
    )
  }
  
  getPrevious(currentIndex: number): string | null {
    if (currentIndex <= 0) return null
    return this.entries[currentIndex - 1].input
  }
  
  getNext(currentIndex: number): string | null {
    if (currentIndex >= this.entries.length - 1) return null
    return this.entries[currentIndex + 1].input
  }
  
  private save() {
    // 保存到 ~/.claude/history.json
    const historyPath = getHistoryPath()
    writeFileSync(historyPath, JSON.stringify(this.entries))
  }
  
  private load() {
    // 从 ~/.claude/history.json 加载
    const historyPath = getHistoryPath()
    if (existsSync(historyPath)) {
      const data = readFileSync(historyPath, 'utf-8')
      this.entries = JSON.parse(data)
    }
  }
}
```

### 历史记录 Hook

```typescript
// hooks/useArrowKeyHistory.tsx
export function useArrowKeyHistory({
  onHistoryPrevious,
  onHistoryNext,
}) {
  const [historyIndex, setHistoryIndex] = useState(-1)
  const [currentInput, setCurrentInput] = useState('')
  
  const historyPrevious = useCallback(() => {
    if (historyIndex === -1) {
      // 保存当前输入
      setCurrentInput(currentInput)
    }
    
    const previous = historyManager.getPrevious(historyIndex)
    if (previous) {
      setHistoryIndex(historyIndex + 1)
      onHistoryPrevious(previous)
    }
  }, [historyIndex, currentInput, onHistoryPrevious])
  
  const historyNext = useCallback(() => {
    const next = historyManager.getNext(historyIndex)
    if (next) {
      setHistoryIndex(historyIndex - 1)
      onHistoryNext(next)
    } else if (historyIndex === 0) {
      // 恢复当前输入
      setHistoryIndex(-1)
      onHistoryNext(currentInput)
    }
  }, [historyIndex, currentInput, onHistoryNext])
  
  return {
    historyPrevious,
    historyNext,
  }
}
```

```typescript
  input: string,
```

```typescript
// components/PromptInput/inputPaste.ts
export function usePasteDetection() {
  const [isPasting, setIsPasting] = useState(false)
  const [pastedText, setPastedText] = useState('')
  const pasteTimeoutRef = useRef<NodeJS.Timeout | null>(null)
  
  const handlePaste = useCallback((text: string) => {
    setIsPasting(true)
    setPastedText(text)
    
    // 清除之前的超时
    if (pasteTimeoutRef.current) {
      clearTimeout(pasteTimeoutRef.current)
    }
    
    // 设置新的超时
    pasteTimeoutRef.current = setTimeout(() => {
      setIsPasting(false)
      setPastedText('')
    }, 100)
  }, [])
  
  return {
    isPasting,
    pastedText,
    handlePaste,
  }
}
```

```typescript
export function usePromptInputPlaceholder({
  isLoading,
  mode,
}) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  const getPlaceholder = () => {
    if (isLoading) {
      return 'Processing...'
    }
    
    switch (mode) {
      case 'command':
        return 'Type a command...'
      case 'search':
        return 'Search...'
      default:
        return 'Type a message...'
    }
  }
  
  return {
    placeholder: getPlaceholder(),
    color: theme.subtle,
  }
}
```
