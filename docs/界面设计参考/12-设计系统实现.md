### 颜色工具函数

```typescript
// components/design-system/color.ts
export function color(
  c: keyof Theme | Color | undefined,
  theme: ThemeName,
  type: ColorType = 'foreground',
): (text: string) => string {
  return text => {
    if (!c) return text
    // 原始颜色值直接使用
    if (c.startsWith('rgb(') || c.startsWith('#') || c.startsWith('ansi256(') || c.startsWith('ansi:')) {
      return colorize(text, c, type)
    }
    // 主题键值查找
    return colorize(text, getTheme(theme)[c as keyof Theme], type)
  }
}
```

### StatusIcon

```typescript
// components/design-system/StatusIcon.tsx
export function StatusIcon({ status }) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  const icons = {
    success: { icon: '✓', color: theme.success },
    error: { icon: '✗', color: theme.error },
    warning: { icon: '!', color: theme.warning },
    info: { icon: '*', color: theme.subtle },
    loading: { icon: '⠋', color: theme.claude },
  }
  
  const { icon, color } = icons[status] || icons.info
  
  return <Text color={color}>{icon}</Text>
}
```

### ProgressBar

```typescript
// components/design-system/ProgressBar.tsx
export function ProgressBar({ value, max, width = 20 }) {
  const [themeName] = useTheme()
  const theme = getTheme(themeName)
  
  const percentage = Math.round((value / max) * 100)
  const filled = Math.round((value / max) * width)
  const empty = width - filled
  
  return (
    <Box>
      <Text color={theme.success}>
        {'█'.repeat(filled)}
      </Text>
      <Text color={theme.inactive}>
        {'░'.repeat(empty)}
      </Text>
      <Text color={theme.subtle}> {percentage}%</Text>
    </Box>
  )
}
```
